    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        double maths,science,english,hindi,social,total,percentage;
        String grade;
        
        maths=Integer.parseInt(txt_maths.getText());
        science=Integer.parseInt(txt_science.getText());
        english=Integer.parseInt(txt_english.getText());
        hindi=Integer.parseInt(txt_hindi.getText());
        social=Integer.parseInt(txt_social.getText());
        
        total=maths+science+english+hindi+social;
        
        percentage=total/5;
        
        if(percentage >= 85)
        {
            grade="A";
        }
        else if(percentage >= 75)
        {
            grade="B";
        }
        else if(percentage >= 65)
        {
            grade="C";
        }
        else if(percentage >= 45)
        {
            grade="D";
        }
        else
        {
            grade="F";
        }
        
        
        txt_total.setText(String.valueOf(total+"/500"));
        txt_percentage.setText(String.valueOf(percentage+"%"));
        txt_grade.setText(grade);
        
        
        
        
        
        
        
    }
